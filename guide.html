The script is based on a number of text and automatically moves the pointer and changes its color.

<script>
 window.onload = function() {
   // Function to update the pointer position and color
   function updatePointer(value) {
     const pointer = document.getElementById('pointer');
     const chart = document.getElementById('chart');      // Retry calculation if dimensions are not available
     if (chart.offsetWidth === 0 || chart.offsetHeight === 0) {
       console.warn('Dimensions not ready, retrying...');
       setTimeout(() => updatePointer(value), 100);
       return;
     }      // Adjust radius to better fit the semi-circle height
     const radius = (chart.offsetHeight * 0.9);      // Calculate the center of the image
     const centerX = chart.offsetWidth / 2;
     const centerY = chart.offsetHeight;      // Map the value 1-100 to π (180 degrees) to 0 (0 degrees)
     const angle = Math.PI - (value / 100) * Math.PI;      // Adjust calculation to better position the pointer
     const x = centerX + radius * Math.cos(angle);
     const y = centerY - radius * Math.sin(angle);      // Update the pointer's position with new offsets
     pointer.style.left = `${x - pointer.offsetWidth / 2}px`; // Center the pointer horizontally
     pointer.style.top = `${y - pointer.offsetHeight / 2}px`; // Center the pointer vertically      // Change the color of the pointer based on the value
     if (value >= 0 && value <= 49) {
       pointer.style.backgroundColor = '#F04438'; // Red
     } else if (value >= 50 && value <= 69) {
       pointer.style.backgroundColor = '#F58F0A'; // Orange
     } else if (value >= 70 && value <= 89) {
       pointer.style.backgroundColor = '#F2C111'; // Yellow
     } else if (value >= 90 && value <= 100) {
       pointer.style.backgroundColor = '#12B76A'; // Green
     }
   }    // Function to handle input changes
   function handleInputChange() {
     const chartValueElement = document.getElementById('chartValue');
     const value = parseInt(chartValueElement.textContent, 10);      if (!isNaN(value) && value >= 1 && value <= 100) {
       updatePointer(value);
     } else {
       console.error('Value must be between 1 and 100');
     }
   }    // Initialize the pointer position
   handleInputChange();    // Set up an observer for dynamic updates
   const observer = new MutationObserver(handleInputChange);
   const chartValueElement = document.getElementById('chartValue');    observer.observe(chartValueElement, {
     childList: true,
     characterData: true,
     subtree: true
   });    // Handle window resize events
   window.addEventListener('resize', handleInputChange);
 };
</script>
